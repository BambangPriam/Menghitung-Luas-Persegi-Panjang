# Menghitung-Luas-Persegi-Panjang
#include <iostream>

using namespace std;

int main()
{
    int p,l,luas;
    cout << "Masukkan panjang persegi panjang: ";
    cin >> p;
    cout <<"Masukkan lebar persegi panjang: ";
    cin>>l;

    luas=p*l;
    cout<<"Luas persegi panjang adalah: "<<luas;
    return 0;
}
